\begin{grammar}
  \bfseries
  <e> ::=
    $x$ \| $c$
    \alt $e$.$a$ \| $e$.$a$ or $e$
    \alt $\λ$$p$.$e$ \| $e$ $e$
    \alt let $x$ = $e$; $\cdots{}$; $x$ = $e$; in $e$
    \alt let rec $r$ = $e$; $\cdots{}$; $r$ = $e$; in $e$
    \alt ($e$, $e$)

  <c> ::= $s$ \| $i$ \| $b$

  <p> ::= $q$ \| $q$@$x$ \| $r$

  <q> ::= ($r$, $r$) \| $q$:\τ

  <r> ::= $x$ | $x$:\τ
\end{grammar}
