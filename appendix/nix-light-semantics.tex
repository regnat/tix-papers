    \sfrac{x}{e}    &= x := e \\

        \quad\text{ if } x = \var(l) \\
    \sfrac{\{ l, \seq{l_i,}{i \in I}, l_m, .. \}}{\{ x = e; \seq{x_j = e_j;}{j \in J}\}}
      &= \sfrac{l}{e};

        \quad\text{ if } x = \var(l) \\
